% Transportation Review Board manuscript formatting
%
% This file largely adapted from http://ite.ce.gatech.edu/?p=681, but meant for
% inclusion in LyX documents. Some elements from the TRB instructions to authors
% may be handled using the LyX UI, including:
% - Font selection (Times Roman)
% - Margins

% abstract
\usepackage[style]{abstract}
\renewcommand{\absnamepos}{flushleft}
\setlength{\absleftindent}{0pt}
\setlength{\absrightindent}{0pt}
\setlength{\abstitleskip}{0pt}
\renewcommand{\abstitlestyle}[1]{\normalfont\bfseries\MakeUppercase{#1}}
\renewcommand{\abstracttextfont}{\normalfont}

% caption:
\usepackage[font=bf,labelsep=quad,justification=RaggedRight]{caption}
\DeclareCaptionLabelFormat{uc}{\MakeUppercase{#1} #2}
\captionsetup{labelformat=uc}
\captionsetup[subfloat]{labelfont=normalsize}

% fancyhdr: Page headers
\renewcommand{\headrulewidth}{0pt}

% lineno: Line numbering
\usepackage[pagewise]{lineno}
\renewcommand\linenumberfont{\normalfont}

%% natbib: Citations and bibliography entries
%\renewcommand{\citenumfont}[1]{\textit{#1}}
%\setcitestyle{round}
%\setlength{\bibsep}{0pt}

\DeclareCiteCommand{\citep}[\mkbibparens]
  {\usebibmacro{cite:init}%
   \usebibmacro{prenote}}
  {\usebibmacro{citeindex}%
   {\itshape\usebibmacro{cite:comp}}}
  {}
  {\usebibmacro{cite:dump}%
   \usebibmacro{postnote}}

% titlesec:
\usepackage[tiny,rm]{titlesec}
\renewcommand*{\refname}{\uppercase{References}}
\titleformat{\section}{\bfseries}{}{0pt}{\uppercase}
\titlespacing*{\section}{0pt}{\baselineskip}{\baselineskip}
\titleformat{\subsection}{\bfseries}{}{0pt}{}
\titlespacing*{\subsection}{0pt}{\baselineskip}{\baselineskip}
\titleformat{\subsubsection}{\itshape}{}{0pt}{}
\titlespacing*{\subsubsection}{0pt}{\baselineskip}{\baselineskip}
