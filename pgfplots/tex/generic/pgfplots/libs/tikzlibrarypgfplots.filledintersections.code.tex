\usetikzlibrary{intersections}

% Defines \pgfplotsretval to contain a modified path including the
% intersections of a preceding call to \pgfintersectionofpaths.
%
% #1 : should be '1' if the FIRST argument of
% \pgfintersectionofpaths should be used and '2' the SECOND
% argument of \pgfintersectionofpaths should be used on input.
% 
%  On output, \pgfplotsretval will be assigned.
\def\pgfplots@insert@intersections@in@path#1{%
	\ifcase#1\relax
		% 0
		\pgfplots@error{illegal argument supplied.}
	\or
		\let\pgfplots@loc@TMPa=\pgf@intersect@path@a
	\or
		\let\pgfplots@loc@TMPa=\pgf@intersect@path@b
	\else
		\pgfplots@error{illegal argument supplied.}
	\fi
	\show\pgfplots@loc@TMPa
}%


\endinput
