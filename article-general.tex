%% Packages
% amsmath: makes \mathbb and other math enhancements available
\usepackage{amsmath}

% appendix: an appendix environment and other goodies
\usepackage[toc]{appendix}

% babel: real Canadian English!
\usepackage[english,canadian]{babel}

% charter: Charter BT 
\usepackage{charter}

% chngcntr: Global sequence of numbers for figures and tables
\usepackage{chngcntr}
  \counterwithout{figure}{section}
  \counterwithout{table}{section}

% ctable: \toprule, \midrule, \bottomrule and other commands for tables
\usepackage{ctable}

% datetime: preferred little-endian date and time formats
\usepackage{datetime}
  % e.g. 01 January 2008
  \newdateformat{ddmmmmyyyydate}{\THEDAY~\monthname[\THEMONTH] \THEYEAR}
  % e.g. 01 Jan 08
  \newdateformat{ddmmmyydate}{\THEDAY~\shortmonthname[\THEMONTH] \THEYEAR}

% fancyhdr: Page headers and footers
\usepackage{fancyhdr}
  \pagestyle{fancy}
  \fancyhfoffset[LE,RO]{0cm}
  \fancyhead[RE,LO]{Paul Natsuo Kishimoto}
  \fancyhead[LE,RO]{\thetitle}
  \renewcommand{\headrulewidth}{0.08em}
  \addtolength{\headheight}{2\baselineskip}
  \cfoot{\bfseries \thepage}

% geometry: margins
\usepackage[hmargin=2.5cm,vmargin={0cm,0.5cm},includeheadfoot]{geometry}

% graphicx: \includegraphics and others
\usepackage{graphicx}

% hyperref: hyperlinks in PDF documents same colour as text and other link properties
\usepackage[colorlinks,citecolor=black,filecolor=black,linkcolor=black,urlcolor=black]{hyperref}
  \urlstyle{same}

% ifpdf: PDF metadata
\usepackage{ifpdf}
  \ifpdf
  \pdfinfo {
    /Author (Paul Natsuo Kishimoto <mail@paul.kishimoto.name>)
    /ModDate (D:\pdfdate) % Sets the PDF 'Modification' time to compile time
  }
  \fi

% inputenc: source files are in UTF-8
\usepackage[utf8]{inputenc}

% longtable: split tables across pages
\usepackage{longtable}

% mathdesign: charter font for math as well
\usepackage[charter]{mathdesign}

% multicol: \multicols environment
\usepackage{multicol}

% natbib: allow some squishing of the bibliography to save space,
% e.g. by using:
%
%   \setlength{\bibsep}{0.25\baselineskip}
%
% ...in the preamble.
\usepackage[square,numbers]{natbib}
  
% nomencl: to use \nomenclature. Note the 'rubber' building system
% for LaTEX (http://www.pps.jussieu.fr/~beffara/soft/rubber/) does
% not yet include support for nomencl, so some support is hacked
% in here.
\usepackage{nomencl}
% rubber: module index
% rubber: onchange $base.nlo "makeindex '$base.nlo' -s nomencl.ist -o '$base.nls'"
% rubber: watch $base.nls
% rubber: clean $base.{}ilg $base.nlo $base.nls $base.out

% setspace: use \setspace{n} to adjust line spacing by a factor
\usepackage{setspace}

% titling: access the document title and date with \thetitle, \thedate
\usepackage{titling}

% titlesec: AIAA-ish section and subsection numbering
\usepackage{titlesec}
  \titleformat{\section}[block]{\filcenter\Large\bfseries}{\thesection.}{4ex}{}{}
  \titleformat{\subsection}[hang]{\large\bfseries}{\thesubsection.}{1ex}{}{}
  \titlespacing{\subsection}{0pt}{*1}{*0}
  \renewcommand{\thesection}{\Roman{section}}
  \renewcommand{\thesubsection}{\Alph{subsection}}

%% End of package includes
