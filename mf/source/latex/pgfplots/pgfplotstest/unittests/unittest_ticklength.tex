\documentclass[a4paper]{article}

\usepackage{pgfplots}
\usepackage{pgfplots.assert}
\pgfplotsset{compat=1.5}

\begin{document}

{
	\pgfplotsset{tickwidth=42pt}
	\pgfplotsassertequalstok{42pt}{\pgfkeysvalueof{/pgfplots/major tick length}}{}
}

{
	\pgfplotsset{major tick length=42pt}
	\pgfplotsassertequalstok{42pt}{\pgfkeysvalueof{/pgfplots/major tick length}}{}
}

{
	\pgfplotsset{subtickwidth=43pt}
	\pgfplotsassertequalstok{43pt}{\pgfkeysvalueof{/pgfplots/minor tick length}}{}
}

{
	\pgfplotsset{minor tick length=42pt}
	\pgfplotsassertequalstok{42pt}{\pgfkeysvalueof{/pgfplots/minor tick length}}{}
}

\begin{tikzpicture}
	\begin{axis}[
		subtickwidth=15pt,
		minor tick num=1,
	]
	\addplot+[samples=2] {x};
	\end{axis}
\end{tikzpicture}

\begin{tikzpicture}
