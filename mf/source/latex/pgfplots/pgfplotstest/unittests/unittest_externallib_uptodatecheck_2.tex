\documentclass{minimal}

\usepackage{pgfplots}

% Without the external library everything compiles.
\usepgfplotslibrary{external}
\tikzexternalize[mode=list and make]

\begin{document}

	No paragraph $\Rightarrow$ no error
	\begin{tikzpicture}
		\draw (0,0) -- +(1,0);
		\draw (0,0) -- +(0,1);
	\end{tikzpicture}

	Paragraph $\Rightarrow$ error
	\begin{tikzpicture}
		\draw (0,0) -- +(1,0);

		\draw (0,0) -- +(0,1);
	\end{tikzpicture}

\end{document}
