\usepackage{pgfplots}
\pgfplotsset{compat=1.3}

\pagestyle{empty}
%\usepgfplotslibrary{clickable}

\def\DoPlot{%
  \addplot+[no marks] expression[domain=1e1:1e5] { x^(.3) * (1 + 1e3*x^(-1)) / (1 + .5e3*x^(-1.3)) };
}

\begin{document}


\begin{tikzpicture}
  \begin{loglogaxis}[log basis x=10,title={log basis x=10}]
    \DoPlot
  \end{loglogaxis}
\end{tikzpicture}
