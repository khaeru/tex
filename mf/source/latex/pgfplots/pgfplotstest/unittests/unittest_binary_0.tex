	% #1: bytes
	% #2: routine
	% #3: numbers
	\long\def\binarytest#1#2#3{%
		\begingroup
		\begingroup
			\escapechar=-1
			\xdef\binarytestname{\string#2}%
		\endgroup
		\section{bytes=#1, \binarytestname}%
		\pgfqkeys{/pgfplots/bin}{bytes=#1}%
		\long\def\binarytestforeach{%
			#2{\num}%
			\num $\mapsto$ \pgfplotsbinaryresult\par
		}%
		\foreach \num in {#3} {%
			\binarytestforeach
		}
		\endgroup
	}%

	}
