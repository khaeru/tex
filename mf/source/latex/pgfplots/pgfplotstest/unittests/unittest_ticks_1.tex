	\long\def\experiment#1#2{
		\subsection{xtick align=#1, ytick align=#2}
		\begin{tikzpicture}
			\begin{axis}[xtick align=#1,ytick align=#2,
				xlabel=$x axis$,
				ylabel=$y axis$,
				xmin=0,xmax=1,
				ymin=629.65,
				ymax=629.70]
			\end{axis}
		\end{tikzpicture}

	}

	\experiment{inside}{outside}
	\experiment{outside}{outside}
