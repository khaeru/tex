\testfile{pgfplotstest.axislines.3d.tex}
{
	\pgfplotsset{
		samples=5,
		domain=-4:4,
		xtick=data,
		ytick=data,
	%	ztick=data,
	}

	\testsection{Boxed}
	\starttikzpicture
		\startaxis[]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture

	\testsection{axis lines=left}
	\starttikzpicture
		\startaxis[
			axis on top,
			axis lines=left,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture

	\testsection{axis lines=right}
	\starttikzpicture
		\startaxis[
			axis on top,
			axis lines=right,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture

	\testsection{axis lines=middle,axis on top}
	\starttikzpicture
		\startaxis[
			axis on top,
			axis lines=middle,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture

	\testsection{Only  axis x line=middle}
	\starttikzpicture
		\startaxis[axis on top,
			axis x line=middle,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture


	\testsection{3d box=complete}
	\starttikzpicture
		\startaxis[
			3d box=complete,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture
	
	\testsubsection{grid lines}
	\starttikzpicture
		\startaxis[
			grid=major,
			3d box=complete,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture

	\testsubsection{grid lines und completeSTAR}
	\starttikzpicture
		\startaxis[
			grid=major,
			3d box=complete*,
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture

	\testsubsection{grid lines und completeSTAR und styles}
	\starttikzpicture
		\startaxis[
			grid=major,
			3d box=complete*,
			3d box foreground style={grid style={red}},
		]
			\addplot3[surf] {x*y};
		\stopaxis
	\stoptikzpicture
}
